dpatch ve pbuilder ile paketleme örneði

$ mkdir -p /dosya/yolu # yeni bir dizin
$ cd /dosya/yolu
$ tar -xvzf /dosya/yolu/gentoo-1.0.2.tar.gz
$ cp -a gentoo-1.0.2 gentoo-1.0.2-orig
$ cd gentoo-1.0.2
$ dh_make -e isim@alan.dom -f /dosya/yolu/gentoo-1.0.2.tar.gz
...  Sorularý yanýtla.
...  Kaynak aðacýný düzelt.
...  Paketleri "dpkg-buildpackage -rfakeroot -us -uc" ile derlemeyi dene.
...  Kaynak paketindekileri düzenleyip derlenebeler hale getir.
...  ../gentoo_1.0.2.orig.tar.gz dosyasýný silme
$ cd ..
$ cp -a gentoo-1.0.2 gentoo-1.0.2-keep # yedekle
$ mv gentoo-1.0.2/debian debian
$ diff -Nru gentoo-1.0.2-orig gentoo-1.0.2 > yama-dosyasý
... Bunu yaparken gentoo-1.0.2 dizininin üstüne yazabilirsiniz.
... Her ihtimale karþý gentoo-1.0.2-keep dizinini silme.
$ mkdir -p debian/patches
$ dpatch patch-template yama/dosyasý \
   -p "01_patchname" "patch-file description" \
   < yama-dosyasý > debian/patches/01_patchname.dpatch
$ cd debian/patches
$ echo 01_patchname.dpatch >00list
$ cd ../..  # /dosya/yolu'na geç.
$ rm -rf gentoo-1.0.2
$ editor debian/rules

debian/rules dosyasýnýn özgün hali: 

config.status: configure
     ./configure --prefix=/usr --mandir=/usr/share
build: config.status
     ${MAKE}
clean:
     $(testdir)
     $(testroot)
     ${MAKE} distclean
     rm -rf debian/imaginary-package debian/files debian/substvars

debian/rules dosyasýný dpatch'ýn kullanabilmesi için bir metin düzenleyici ile þu hale getirin: 

config.status: patch configure
     ./configure --prefix=/usr --mandir=/usr/share
build: config.status
     ${MAKE}
clean: clean-patched unpatch
clean-patched:
     $(testdir)
     $(testroot)
     ${MAKE} distclean
     rm -rf debian/imaginary-package debian/files debian/substvars
patch: patch-stamp
patch-stamp:
     dpatch apply-all
     dpatch call-all -a=pkg-info >patch-stamp

unpatch:
     dpatch deapply-all
     rm -rf patch-stamp debian/patched

dpatch sistemi ile kaynak aðacýný yeniden paketlemek için artýk hazýrsýnýz. 

$ tar -xvzf gentoo_1.0.2.orig.tar.gz
$ cp -a debian/ gentoo-1.0.2/debian
$ cd gentoo-1.0.2
$ sudo pbuilder update
$ pdebuild
$ cd /var/cache/pbuilder/result/
$ dupload -tnm_target gentoo_1.0.2-1_i386.changes
