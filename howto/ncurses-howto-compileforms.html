#include <form.h>

int main()
{ FIELD *field[3];
  FORM  *my_form;
  int ch;

  /* Curses kipini ilklendir*/
  initscr();
  cbreak();
  noecho();
  keypad(stdscr, TRUE);

  /* Alanlarý ilklendir */
  field[0] = new_field(1, 10, 4, 18, 0, 0);
  field[1] = new_field(1, 10, 6, 18, 0, 0);
  field[2] = NULL;

  /* alan seçeneklerini ayarla*/
  set_field_back(field[0], A_UNDERLINE);  /* Seçenek için bir satýr yazýn  */
  field_opts_off(field[0], O_AUTOSKIP);   /* Eðer alan dolduyda            */
                                          /* alan içerisinde ilerleme      */
  set_field_back(field[1], A_UNDERLINE);
  field_opts_off(field[1], O_AUTOSKIP);

  /* Form2u oluþtur ve ekrana yaz */
  my_form = new_form(field);
  post_form(my_form);
  refresh();

  mvprintw(4, 10, "Deðer 1:");
  mvprintw(6, 10, "Deðer 2:");
  refresh();

  /* Kullanýcý isteklerinde hareket edin */
  while((ch = getch()) != KEY_F(1))
  { switch(ch)
    { case KEY_DOWN:
        /* Sonraki alana gidin */
        form_driver(my_form, REQ_NEXT_FIELD);
        /* O anki önbelleðin sonuna gidin */
        /* Son karakterde döngüden çýk */
        form_driver(my_form, REQ_END_LINE);
        break;
      case KEY_UP:
        /* Bir önceki alana git */
        form_driver(my_form, REQ_PREV_FIELD);
        form_driver(my_form, REQ_END_LINE);
        break;
      default:
        /* Eðer normal bir karakterse */
        /* Ekrana yazýlýr             */
        form_driver(my_form, ch);
        break;
    }
  }

  /* Formu ekrandan sil ve ayrýlan bellek alanýný geri ver */
  unpost_form(my_form);
  free_form(my_form);
  free_field(field[0]);
  free_field(field[1]);

  endwin();
  return 0;
}

Beþinci parametre, ekranda gözükmeyen satýr sayýsý alanýn ne kadarýnýn gözükeceðini belirtir. Eðer sýfýrsa, tüm alan gösterilir, aksi halde kullanýcý alanýn görüntülenmeyen alanlarýna taþtýðýnda form kaydýrýlabilir olacaktýr. Form kütüphanesi kullanýcýnýn girdiði verilerin saklanmasý için her alan için bir önbellek tahsis eder. new_field()'in son parametresini kullanarak ilave ek bellekler tahsis etmesini saðlayabiliriz. Ýstediðiniz herhangi bir amaç için kullanýlabilir.
