# komutlarý için kullanýlýr. Bu sýnamalar selamlaþma kabul ya da
# red edilinceye kadar sýrayla yapýlýr.


acl_helo:

  # bilgisini kaydediyoruz.
  warn
    set acl_m2  = $tod_epoch


  # Posta yerel SMTP üzerinden alýnmýþsa (yani, TCP/IP baðlantýsý
  # ile gelmiyorsa), kabul ediyoruz. Bunu boþ bir konak alanýný
  # sýnayarak yapacaðýz. Ayrýca, postalarýný rölelediðimiz
  # konaklardan gelen postalarý da kabul edeceðiz.

  accept

  # bir red iletisi ve $acl_c1'e bir günlük iletisi kaydedeceðiz. Bunlarý
  # sonra bir "deny" deyiminde kullanacaðýz ve bu sýrada göndericiyi 20 saniye
  # bekleteceðiz.

  warn

    set acl_c0  = Message was delivered by ratware
    set acl_c1  = remote host used IP address in HELO/EHLO greeting


  # Baðlanan konak selamlaþma sýrasýnda bizim isimlerimizden birini
  # belirtmiþse ayný iþleme tabi tutuyoruz.

  warn
    condition   = ${if match_domain{$sender_helo_name}\
                   {$primary_hostname:+local_domains:+relay_to_domains}\
                   {true}{false}}
    set acl_c0  = Message was delivered by ratware
    set acl_c1  = remote host used our name in HELO/EHLO greeting.


  # HELO doðrulamasý baþarýsýz olmuþsa, acl_c1'e bir uyarý iletisi
  # kaydediyoruz. Bu iletiyi daha sonra posta baþlýðýna ekleyeceðiz.
  # Varlýðý bize geciktirme uygulayacaðýmýzý belirtecek.
  #
  warn
    condition   = ${if !def:acl_c1 {true}{false}}
    !verify     = helo
    set acl_c1  = X-HELO-Warning: Remote host $sender_host_address \
                  ${if def:sender_host_name {($sender_host_name) }}\
                  incorrectly presented itself as $sender_helo_name
    log_message = remote host presented unverifiable HELO/EHLO greeting.


  # Selamlaþmayý kabul ediyoruz, fakat $acl_c1'de evvelce kaydedilmiþ
  # bir ileti varsa, göndericiyi 20 saniye dolana kadar bekletiyoruz.
  accept
    set acl_m2  = ${if def:acl_c1 {${eval:20 + $acl_m2 - $tod_epoch}}{0}}
    delay       = ${if >{$acl_m2}{0}{$acl_m2}{0}}s
