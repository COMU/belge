Örnek 5.9. Basit bir scanw örneði 

#include <ncurses.h>          /* ncurses.h, stdio.h da ekler */
#include <string.h>

int main()
{
 char ileti[]="Bir dizge giriniz: ";  /* ekranda gözükecek mesaj */
 char dizge[80];
 int sat,sut;                     /* ekranýn satýr ve sütün sayýsýný  *
                                   * tutmak için */
 initscr();                       /* curses kipine baþlama */
 getmaxyx(stdscr,sat,sut);        /* satýr ve sütün satýsýný elde et */
 mvprintw(sat/2, (sut-strlen(ileti))/2, "%s", ileti);
                                  /* iletiyi ekranýn ortasýna yaz */
 getstr(dizge);
 mvprintw(LINES - 2, 0, "Þunu girdiniz: %s", dizge);
 getch();
 endwin();

 return 0;
}
