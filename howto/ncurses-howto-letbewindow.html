Bana bir Pencere göster !!!

Eðer bir pencere oluþturuldu ve biz de göremiyorsak bu komik bir durumdur. Eðlenceli kýsým pencereyi göstermekle baþlar. box() iþlevi pencere etrafýna bir çerçeve çizmek için kullanýlýr. Bu iþlevleri aþaðýdaki örnekte daha ayrýntýlý inceleyelim.

Örnek 5.12. Pencere Çerçevesi örneði

#include <ncurses.h>

WINDOW *create_newwin(int height, int width, int starty, int startx);
void destroy_win(WINDOW *local_win);

int main(int argc, char *argv[])
{ WINDOW *my_win;
  int startx, starty, width, height;
  int ch;

  initscr();                      /* Curses kipini ilklendirme    */
  cbreak();                       /* Satýr önbelleði etkisiz,     *
                                   * herþeyi bana gönder          */
  keypad(stdscr, TRUE);           /* Þu þeker F1'e ihtiyacým var  */

  height = 3;
  width = 10;
  starty = (LINES - height) / 2;  /* Pencerenin ortasýna yazmak için */
  startx = (COLS - width) / 2;    /* hesaplama yapýlýyor             */
  printw("Press F1 to exit");
  refresh();
  my_win = create_newwin(height, width, starty, startx);

  while((ch = getch()) != KEY_F(1))
  { switch(ch)
    { case KEY_LEFT:
        destroy_win(my_win);
        my_win = create_newwin(height, width, starty,--startx);
        break;
      case KEY_RIGHT:
        destroy_win(my_win);
        my_win = create_newwin(height, width, starty,++startx);
        break;
      case KEY_UP:
        destroy_win(my_win);
        my_win = create_newwin(height, width, --starty,startx);
        break;
      case KEY_DOWN:
        destroy_win(my_win);
        my_win = create_newwin(height, width, ++starty,startx);
        break;
    }
  }

  endwin();                       /* Curses kipi sonu */
  return 0;
}

WINDOW *create_newwin(int height, int width, int starty, int startx)
{ WINDOW *local_win;

  local_win = newwin(height, width, starty, startx);
  box(local_win, 0 , 0);          /* 0, 0 yatay ve düþey çizgiler    *
                                   * için öntanýmlý karakteri verir  */
  wrefresh(local_win);            /* Kutuyu gösterir                 */

  return local_win;
}

void destroy_win(WINDOW *local_win)
{
  /* box(local_win, ' ', ' '); : Bu istenen pencere silme sonucunu
    * vermeyecektir. Pencerenin dört köþesi ve
    * çirkin küçük bir kýsmý kalacaktýr.
    */
  wborder(local_win, ' ', ' ', ' ',' ',' ',' ',' ',' ');
  /* Alýnan parametreler þu þekildedir
    * 1. win: üzerinde çalýþýlacak pencere
    * 2. ls: pencerenin sol kenarý için kullanýlacak karakter
    * 3. rs: pencerenin sað kenarý için kullanýlacak karakter
    * 4. ts: pencerenin üst kenarý için kullanýlacak karakter
    * 5. bs: pencerenin alt kenarý için kullanýlacak karakter
    * 6. tl: pencerenin sol üst köþesi için kullanýlacak karakter
    * 7. tr: pencerenin sað üst köþesi için kullanýlacak karakter
    * 8. bl: pencerenin sol alt köþesi için kullanýlacak karakter
    * 9. br: pencerenin sað alt köþesi için kullanýlacak karakter
    */
  wrefresh(local_win);
  delwin(local_win);
}
