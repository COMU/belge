acl_mail_from

# Bu eriþim denetim listesi gelen bir SMTP aktarýmýnda MAIL FROM:
# komutu için kullanýlýr. Bu sýnamalar gönderici adresi kabul
# ya da red edilinceye kadar sýrayla yapýlýr.
#

acl_mail_from:
  # Gecikme uygularken baþlangýç olarak kullanmak üzere o anki zaman
  # bilgisini kaydediyoruz.
  warn
    set acl_m2  = $tod_epoch


  # Posta yerel SMTP üzerinden alýnmýþsa (yani, TCP/IP baðlantýsý
  # ile gelmiyorsa), kabul ediyoruz. Bunu boþ bir konak alanýný
  # sýnayarak yapacaðýz. Ayrýca, postalarýný rölelediðimiz
  # konaklardan gelen postalarý da kabul edeceðiz.
  #
  # Alýcý doðrulamasýný burada atlýyoruz, çünkü çoðu durumda
  # istemciler kullanýcýlarýn posta istemcileridir ve SMTP
  # hata iletileri ile ne yapacaklarýný bilmezler.
  #
  accept
    hosts     = : +relay_from_hosts


  # Ýleti, kimlik kanýtlamasý yapýlan bir baðlantý üzerinden
  # geliyorsa kabul ediyoruz. Yine, bu iletiler kullanýcýlarýn posta
  # istemcilerinden geldiklerinden alýcý doðrulamasý yapmýyoruz.
  #
  accept
    authenticated = *


  # Bu SMPT aktarýmý sýrasýnda yapýlacak her teslimata uygulanacak red
  # ve/veya uyarý iletileri varsa, bunlar $acl_c0 ve $acl_c1 deðiþkenlerinde
  # kayýtlýdýr. Bunlarý iletiye özel deðiþkenler olan $acl_m{0,1}'a
  # kopyalayýp, $acl_m1'deki uyarý iletisini ileti baþlýðýna ekleyeceðiz.
  # (Bir red durumunda, $acl_m1 artýk bir günlük kaydý içeriyor olacak,
  # ama iletiyi bununla deðil baþlýðýndaki kayýtla reddedeceðiz.)
  #
  warn
    set acl_m0  = $acl_c0
    set acl_m1  = $acl_c1
    message     = $acl_c1


  # Gönderici bir HELO/EHLO selamlaþmasý yapmamýþsa, $acl_m0'e bir red,
  # ve $acl_m1'e bir günlük iletisi kaydedeceðiz. Bunlarý sonra bir
  # "deny" deyiminde kullanacaðýz ve bu sýrada göndericiyi 20 saniye
  # bekleteceðiz.
  #
  warn
    condition   = ${if def:sender_helo_name {0}{1}}
    set acl_m0  = Message was delivered by ratware
    set acl_m1  = remote host did not present HELO/EHLO greeting.


  # Gönderici adresi doðrulanamazsa, $acl_m1'e bir uyarý iletisi
  # kaydedeceðiz ve bunu ileti baþlýðýna ekleyeceðiz.
  # Varlýðý bize geciktirme uygulayacaðýmýzý belirtecek.
  #
  # "callout" seçeneðini isterseniz silebilirsiniz. Özellikle, postanýzý
  # doðrudan deðil de göstermelik sunucu (smarthost) olarak
  # gönderiyorsanýz, bu seçenek anlamsýz olacaktýr.
  #
  warn
    condition   = ${if !def:acl_m1 {true}{false}}
    !verify     = sender/callout
    set acl_m1  = Invalid sender <$sender_address>
    message     = X-Sender-Verify-Failed: $acl_m1
    log_message = $acl_m1


  # Göndericiyi kabul ediyoruz, fakat $acl_c1'de evvelce kaydedilmiþ
  # bir ileti varsa, göndericiyi 20 saniye dolana kadar bekletiyoruz.
  accept
    set acl_m2  = ${if def:acl_c1 {${eval:20 + $acl_m2 - $tod_epoch}}{0}}
    delay       = ${if >{$acl_m2}{0}{$acl_m2}{0}}s
