int main(int argc, char *argv[])

  while((ch = fgetc(fp)) != EOF)        /* dosyayý sonuna kadar oku  */
  {
    getyx(stdscr, y, x);                /* o anki imleç konumunu al  */
    if(y == (row - 1))                  /* ekranýn sonunda mýyýz?    */
    {

                              /* kullanýcýya bir tuþa basmasýný söyle */

                                         * koyu göstermeyi etkinleþtir */

      getyx(stdscr, y, x);              /* imlecin o anki konumunu al */

  fclose(fp);
  return 0;
}
